# so_long

To run on Ubuntu, place the minilibx-linux folder inside the lib directory and adjust the key codes as follows;

# define KEY_A 97
# define KEY_S 115
# define KEY_D 100
# define KEY_W 119
# define KEY_R 114
# define KEY_ONE 49
# define KEY_TWO 50
# define KEY_THREE 51
# define KEY_FOUR 52
# define KEY_ESC 65307
# define KEY_LEFT 65361
# define KEY_RIGHT 65363
# define KEY_DOWN 65364
# define KEY_UP 65362

# define DIR_LEFT 	4
# define DIR_RIGHT 	6
# define DIR_UP 	8
# define DIR_DOWN 	2
